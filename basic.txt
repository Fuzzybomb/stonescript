//Action Animation Cancel
?item.left.state = 3
  equipL *0 wand //trash
  equipL @item.left@
?item.right.state = 3
  equipR *0 shield //trash
  equip @item.right@

/* ------- Deadwood ------- */
?loc = deadwood
  ?hp < 7
    activate potion
  
  //No Enemies
  ?foe.count < 1
    equipL ouroboros
    equipR shield
    ?pickup.distance < 10
      equipL star
  
  //Mobs
  :?foe ! boss
    >@foe.name@
    ?foe = scarab & foe.distance <= 16 & 
    ^ foe.distance >= 11
      equipR dashing
    :?foe = huge_mosquito
      equipR crossbow
      equipL ouroboros
    :?pickup.distance < 10
      equipL star
    :?foe.distance > 8 & hp < (maxhp - 10)
      equipL ouroboros
    :
      equipL sword
      equipR shield
    
  //Bossing
  :?foe = boss
    ?foe.distance >= 11 & foe.distance <= 16
      equipR dashing
    :?foe.debuffs.count = 0
      equipR dp
    :
      equipL sword
      equipR crossbow
